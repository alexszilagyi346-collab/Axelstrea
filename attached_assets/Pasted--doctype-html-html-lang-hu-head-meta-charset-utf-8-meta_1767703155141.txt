  <script>
    const videoUrlInput = document.getElementById('videoUrl');
    const posterInput = document.getElementById('posterUrl');
    const loadBtn = document.getElementById('loadBtn');
    const testCORSBtn = document.getElementById('testCORS');
    const useIframeBtn = document.getElementById('useIframe');
    const downloadBtn = document.getElementById('downloadBtn');

    const videoPlayer = document.getElementById('videoPlayer');
    const iframeContainer = document.getElementById('iframeContainer');
    const videoIframe = document.getElementById('videoIframe');

    const statusMessage = document.getElementById('statusMessage');
    const headersOutput = document.getElementById('headersOutput');
    const logOutput = document.getElementById('logOutput');

    function log(msg){
      const time = new Date().toLocaleTimeString();
      logOutput.textContent = `[${time}] ${msg}\n` + logOutput.textContent;
      console.log(msg);
    }

    function setStatus(msg, isError=false){
      statusMessage.textContent = 'Állapot: ' + msg;
      statusMessage.style.color = isError ? '#ffb4b4' : '';
    }

    function resetUI(){
      iframeContainer.style.display = 'none';
      videoIframe.src = '';
      videoPlayer.pause();
      videoPlayer.removeAttribute('src');
      while (videoPlayer.firstChild) videoPlayer.removeChild(videoPlayer.firstChild);
      headersOutput.textContent = 'Nincsenek adatok.';
      downloadBtn.style.display = 'none';
    }

    async function tryLoadVideo(url){
      resetUI();
      setStatus('Betöltés: ellenőrzés folyamatban…');
      log('Próbálkozás betöltéssel: ' + url);

      // beállítjuk a poster, ha van
      const poster = posterInput.value.trim();
      if(poster) videoPlayer.setAttribute('poster', poster);
      else videoPlayer.removeAttribute('poster');

      // Alapvető fetch HEAD ellenőrzés (CORS fejlécek lekérése)
      try{
        const headResp = await fetch(url, { method: 'HEAD' });
        log('HEAD státusz: ' + headResp.status);
        const ct = headResp.headers.get('content-type') || '–';
        const cl = headResp.headers.get('content-length') || '–';
        const acceptRanges = headResp.headers.get('accept-ranges') || '–';
        headersOutput.textContent = `status: ${headResp.status}\ncontent-type: ${ct}\ncontent-length: ${cl}\naccept-ranges: ${acceptRanges}\n\nfull headers:\n` + [...headResp.headers.entries()].map(h=>h.join(': ')).join('\n');

        if (headResp.ok && ct.startsWith('video/')){
          // próbáljuk meg betenni a source elemet
          const source = document.createElement('source');
          source.src = url;
          source.type = ct;
          videoPlayer.appendChild(source);
          videoPlayer.load();
          setStatus('A videó betöltve — próbáld lejátszani.');
          downloadBtn.style.display = 'inline-block';
          downloadBtn.href = url;
          log('Videó betöltve a <video> elembe.');
        } else {
          setStatus('A HEAD válasz nem video tartalmat jelez, iframe fallback javasolt.', true);
          log('HEAD nem video (vagy hibás). Használd az iframe-t fallbackként.');
          iframeContainer.style.display = 'block';
          videoIframe.src = url;
        }
      } catch (err){
        // Ha a HEAD fetch CORS miatt elutasítva van, megpróbáljuk sima set src-re tenni
        log('HEAD fetch hiba: ' + err);
        setStatus('CORS vagy hálózati hiba a HEAD kéréssel — megpróbáljuk a közvetlen <video> betöltést.', true);
        try {
          const source = document.createElement('source');
          // jelöljük meg, hogy nem tudjuk a content-type-ot
          source.src = url;
          videoPlayer.appendChild(source);
          videoPlayer.load();
          setStatus('Próbálkozás közvetlen <video> betöltéssel. Ha nem működik, használd az iframe fallbacket.', true);
          downloadBtn.style.display = 'inline-block';
          downloadBtn.href = url;
        } catch(e){
          log('Közvetlen betöltés sikertelen: ' + e);
          iframeContainer.style.display = 'block';
          videoIframe.src = url;
          setStatus('Nem sikerült betölteni a videót, iframe fallback megjelenítve.', true);
        }
      }
    }

    loadBtn.addEventListener('click', () => {
      const url = videoUrlInput.value.trim();
      if(!url){ setStatus('Adj meg egy videó URL-t.', true); return; }
      tryLoadVideo(url);
    });

    testCORSBtn.addEventListener('click', async () => {
      const url = videoUrlInput.value.trim();
      if(!url){ setStatus('Adj meg egy videó URL-t a CORS teszthez.', true); return; }
      setStatus('CORS ellenőrzés: próbálok HEAD-et küldeni…');
      log('CORS teszt indítva a ' + url);
      try {
        const resp = await fetch(url, { method: 'HEAD' });
        setStatus('HEAD sikeres: státusz ' + resp.status);
        headersOutput.textContent = [...resp.headers.entries()].map(h=>h.join(': ')).join('\n');
        log('HEAD sikeres, header-ek kiírva.');
      } catch (err){
        setStatus('CORS vagy hálózati hiba a HEAD kéréssel — nézd meg a konzolt.', true);
        log('HEAD kérés hiba: ' + err);
      }
    });

    useIframeBtn.addEventListener('click', () => {
      const url = videoUrlInput.value.trim();
      if(!url){ setStatus('Adj meg egy URL-t az iframe-hez.', true); return; }
      iframeContainer.style.display = 'block';
      videoIframe.src = url;
      setStatus('Iframe fallback megjelenítve.');
      log('Iframe fallback engedélyezve: ' + url);
    });

    // gyors példaérték betöltéshez (távolítható)
    // videoUrlInput.value = 'https://storage.googleapis.com/gtv-videos-bucket/sample/BigBuckBunny.mp4';
